#!/bin/bash

max=100; min=1
for i in $(ls 1$)
do
     for v in $(cat $1$i)
     do
          age=$(echo $v | cut -d ":" -f2)
           #echo $age
           if [ $age -lt $max ]
           then
                  max=$age
            elif [ $age -gt $max ]
            then
                     min=$age
            fi
     done
done
echo $max
echo $min
