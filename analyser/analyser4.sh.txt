#!/bin/bash

youngest_age=100
youngest_name=""
youngest_email=""

oldest_age=0
oldest_name=""
oldest_email=""

for file in $(ls $1/*.txt)
do
    name=$(grep "Имя:" $file | cut -d ":" -f2)
    age=$(grep "Возраст:" $file | cut -d ":" -f2)
    email=$(grep "Email:" $file | cut -d ":" -f2)
    
    if [ $age -lt $youngest_age ]
    then
        youngest_age=$age
        youngest_name=$name
        youngest_email=$email
    fi
    
    if [ $age -gt $oldest_age ]
    then
        oldest_age=$age
        oldest_name=$name
        oldest_email=$email
    fi
done

echo "Самый молодой: Имя:$youngest_name, Возраст:$youngest_age, Email:$youngest_email"
echo "Самый старший: Имя:$oldest_name, Возраст:$oldest_age, Email:$oldest_email"
